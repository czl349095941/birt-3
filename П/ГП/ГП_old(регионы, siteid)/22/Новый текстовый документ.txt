SELECT    cls.classstructureid nr1, 
           cls.description    meri, 
           locations.description pch, 
           asset.description stan, 
case  
when asset.itemnum in  ('цооя', 'осрэо') then spec1.alnvalue  
when asset.itemnum = 'ярюмосрэ' THEN spec2.alnvalue  
when  asset.itemnum = 'яро' then spec3.alnvalue 
end nPut,  
nvl(pm.obrab, '1') obrab , 
 cast (longdescription.ldtext as varchar (1000)) ldtext
 FROM  maximo.CLASSSTRUCTURE cls  
  left JOIN maximo.masterpm ON masterpm.namerab1 = cls.classstructureid 
  LEFT JOIN maximo.pm ON  masterpm.masterpmnum = pm.masterpm 
  LEFT JOIN maximo.asset  ON  pm.assetnum = asset.assetnum 
	left join maximo.assetfeaturespec s on asset.assetnum = s.assetnum and asset.siteid = s.siteid and s.ASSETATTRID = 'PCH_ECH' 
  LEFT JOIN maximo.locations  ON  masterpm.location=locations.location 
  left join maximo.assetlocrelation al on al.targetassetnum = pm.assetnum and al.assetrelationnum in ('цкосрэ_осрэо', 'цкосрэ_цооя') and  al.siteid = pm.siteid 
  left join maximo.assetspec spec1 on (al.sourceassetnum = spec1.assetnum AND spec1.ASSETATTRID = 'NOM2' and  al.siteid = spec1.siteid)     
  left outer join maximo.assetspec spec2 on (maximo.asset.assetnum = spec2.assetnum AND spec2.ASSETATTRID = 'PUTST_NOM')  
  left outer join maximo.assetspec spec3 on (Al.SOURCEASSETNUM = spec3.assetnum AND spec3.ASSETATTRID = 'STP_NOM') 
  left join MAXIMO.longdescription on (pm.pmuid = longdescription.ldkey  and longdescription.ldownertable = 'PM')
 WHERE     cls.classstructureid IN ( '090000070883', 
                                 '090000070884' , 
                                 '090000070885' , 
                                 '090000070886' , 
                                 '090000070887' , 
                                 '090000070888' , 
                                 '090000070889' , 
                                 '090000070890' , 
                                 '090000070891' , 
                                 '090000070892' , 
                                 '090000070893') 
 AND       masterpm.year      = 2013
 AND       locations.siteid  = '24' --and locations.location='PRED_3732'
